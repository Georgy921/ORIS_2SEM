DROP TABLE IF EXISTS tours;

CREATE TABLE tours (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    arrival_city VARCHAR(100) NOT NULL,
    departure_city VARCHAR NOT NULL,
    departure_date VARCHAR NOT NULL,
    nights_count INTEGER NOT NULL CHECK (nights_count > 0),
    adults_count INTEGER NOT NULL CHECK (adults_count > 0),
    tour_price DECIMAL(10, 2) NOT NULL CHECK (tour_price >= 0),
    image_url VARCHAR(500), 
    bonus INTEGER NOT NULL CHECK (bonus > 0),
    wifi VARCHAR(100),
    distance_to_airport INTEGER,
    general_description TEXT,
    has_kids_club BOOL,
    hotel_name VARCHAR(255),
    has_aquapark BOOL,
    has_spa BOOL,
    room_type VARCHAR(50),
    popular_filters VARCHAR(50),
    region VARCHAR(50),
    rating INTEGER DEFAULT 5 CHECK (rating BETWEEN 1 AND 5),
    meal_plan VARCHAR(50) DEFAULT 'all-inclusive'
);
